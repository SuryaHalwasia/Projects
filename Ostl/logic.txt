for s1 in keyword.kwlist:
        idx = '1.0'
        while 1:
            idx = e1.search(s1, idx, nocase=1, stopindex=END)
            if not idx:
                break
            lastidx = '%s+%dc' % (idx, len(s1))
            e1.tag_add('found', idx, lastidx)
            idx = lastidx
        e1.tag_config('found', foreground='purple')
